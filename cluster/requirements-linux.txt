# requirements-linux.txt — conda env for the random_logical cluster pipeline (RedHat/Linux)
# =============================================================================
# Scope: ONLY the four cluster stages and everything they import:
#   1. _nemo.py --sdg-logical      (imports the aisa/ package — the heavy one)
#   2. extract_artifacts.py        (standalone)
#   3. generate-qa.py              (standalone)
#   4. self-check/self-check-qa.py (standalone)
# Versions are pinned to the set verified end-to-end by the local smoke run.
# Transitive deps (httpx, anyio, tenacity, tokenizers, safetensors, the google-*
# stack, …) resolve automatically from the pins below — not listed.
#
# Setup on the cluster (login node):
#   conda create -n qac-spec python=3.12 -y      # MUST be >=3.11 (stdlib `tomllib`
#                                                #   is imported unconditionally)
#   conda activate qac-spec
#   # torch: install the CPU build FIRST (Python never touches the GPU — Ollama
#   # serves the models separately; torch is only imported at module load and
#   # all-MiniLM is never actually run in random_logical). Avoids a multi-GB
#   # CUDA wheel + CUDA/driver mismatch on RedHat:
#   pip install torch==2.7.1 --index-url https://download.pytorch.org/whl/cpu
#   pip install -r cluster/requirements-linux.txt
#
# NOT pip packages (do not add here): the Ollama models gpt-oss:120b /
# nomic-embed-text:v1.5 (pulled by the slurm job / pull_models.sh) and the HF
# weights sentence-transformers/all-MiniLM-L6-v2 (a model cache — see SRS §6.4).
#
# Deliberately EXCLUDED (imported only by the larger sibling project / off-path
# eval scripts, NOT by stages 1–4): faiss-cpu, qdrant-client, datasets, docling,
# vllm, spacy, streamlit, easyocr, anthropic, …  Keeps this env lean.
# =============================================================================
